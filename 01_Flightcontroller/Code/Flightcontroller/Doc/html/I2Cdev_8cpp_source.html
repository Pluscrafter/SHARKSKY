84  // 01101001 read byte
85  // 76543210 bit numbers

150 bool I2Cdev::writeBit(uint8_t devAddr, uint8_t regAddr, uint8_t bitNum, uint8_t data) {
151 
152  //first reading registery value
153  sendBuf[0] = regAddr;
154  uint8_t buffer;
155  while(HAL_I2C_Mem_Read(&hi2c1,devAddr,regAddr,I2C_MEMADD_SIZE_8BIT,&buffer,1,I2CDLY) != HAL_OK){
156  MX_I2C1_Init();
157  }
158  uint8_t response = BCM2835_I2C_REASON_OK;
159  if ( response == BCM2835_I2C_REASON_OK ) {
160  uint8_t b = buffer ;
161  b = (data != 0) ? (b | (1 << bitNum)) : (b & ~(1 << bitNum));
162  sendBuf[1] = b ;
163  while(HAL_I2C_Mem_Write(&hi2c1,devAddr,regAddr, I2C_MEMADD_SIZE_8BIT, (uint8_t*)&b, 1,I2CDLY) != HAL_OK){
164  MX_I2C1_Init();
165  }
166  response = BCM2835_I2C_REASON_OK;
167  }
168  return response == BCM2835_I2C_REASON_OK;
169 }

179 bool I2Cdev::writeBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t data) {
180  // 010 value to write
181  // 76543210 bit numbers
182  // xxx args: bitStart=4, length=3
183  // 00011100 mask byte
184  // 10101111 original value (sample)
185  // 10100011 original & ~mask
186  // 10101011 masked | value
187 
188  //first reading registery value
189  sendBuf[0] = regAddr;
190  uint8_t buffer;
191  while(HAL_I2C_Mem_Read(&hi2c1,devAddr,regAddr,I2C_MEMADD_SIZE_8BIT,&buffer,1,I2CDLY) != HAL_OK){
192  MX_I2C1_Init();
193  }
194  uint8_t response = BCM2835_I2C_REASON_OK;
195  if ( response == BCM2835_I2C_REASON_OK ) {
196  uint8_t b = buffer;
197  uint8_t mask = ((1 << length) - 1) << (bitStart - length + 1);
198  data <<= (bitStart - length + 1); // shift data into correct position
199  data &= mask; // zero all non-important bits in data
200  b &= ~(mask); // zero all important bits in existing byte
201  b |= data; // combine data with existing byte
202  sendBuf[1] = b ;
203  while(HAL_I2C_Mem_Write(&hi2c1,devAddr,regAddr, I2C_MEMADD_SIZE_8BIT, (uint8_t*)&b, 1,I2CDLY) != HAL_OK){
204  MX_I2C1_Init();
205  }
206  response = BCM2835_I2C_REASON_OK;
207  }
208  return response == BCM2835_I2C_REASON_OK;
209 }
