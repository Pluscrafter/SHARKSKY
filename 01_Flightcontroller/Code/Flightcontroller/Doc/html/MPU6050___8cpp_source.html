33 void MPU6050::Init(){
34  //reset Sensor
35  uint8_t buffer[3] = {0x80,0x07,0x00};
36  HAL_I2C_Mem_Write(&hi2c1,address,PWR_MGMT_1, I2C_MEMADD_SIZE_8BIT, (uint8_t*)&buffer[0], 1,DELAY);
37  HAL_Delay(50);
38  HAL_I2C_Mem_Write(&hi2c1,address,SMPLRT_DIV, I2C_MEMADD_SIZE_8BIT, (uint8_t*)&buffer[1], 1,DELAY);
39  HAL_Delay(50);
40  HAL_I2C_Mem_Write(&hi2c1,address,PWR_MGMT_1, I2C_MEMADD_SIZE_8BIT, (uint8_t*)&buffer[2], 1,DELAY);
41  HAL_Delay(50);
42  //config
43  Config();
44  HAL_Delay(50);
45  //set res to 500ï¿½/s
46  SetGyroResolution();
47  HAL_Delay(50);
48  //set res to 4g/s
49  SetACCLResolution();
50  HAL_Delay(50);
51 }
