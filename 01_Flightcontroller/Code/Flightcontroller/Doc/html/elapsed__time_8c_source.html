56 void elapsed_time_init (void)
57 {
58  uint32_t i;
59 
60 
61  if (ARM_CM_DWT_CTRL != 0) { // See if DWT is available
62  ARM_CM_DEMCR |= 1 << 24; // Set bit 24
63  ARM_CM_DWT_CYCCNT = 0;
64  ARM_CM_DWT_CTRL |= 1 << 0; // Set bit 0
65  }
66  for (i = 0; i < ELAPSED_TIME_MAX_SECTIONS; i++) {
67  elapsed_time_clr(i);
68  }
69 }

84 * STOP THE MEASUREMENT OF A CODE SECTION AND COMPUTE STATS
85 ********************************************************************************
86 */
87 
88 void elapsed_time_stop (uint32_t i)
89 {
90  uint32_t stop;
91  ELAPSED_TIME *p_tbl;
92 
93 
94  stop = ARM_CM_DWT_CYCCNT;
95  p_tbl = &elapsed_time_tbl[i];
96  p_tbl->current = stop - p_tbl->start;
97  if (p_tbl->max < p_tbl->current) {
98  p_tbl->max = p_tbl->current;
99  }
100  if (p_tbl->min > p_tbl->current) {
101  p_tbl->min = p_tbl->current;
102  }
103 }
